Hourly Application Volume last 24h

API Latency by Endpoint P50 / P95 / P99 in ms · TRANSACTIONLOG.LATENCY_MS

SELECT a.APINAME, PERCENTILE_CONT(0.50) WITHIN GROUP (ORDER BY t.LATENCY_MS) OVER (PARTITION BY a.APINAME) AS p50, PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY t.LATENCY_MS) OVER (PARTITION BY a.APINAME) AS p95, PERCENTILE_CONT(0.99) WITHIN GROUP (ORDER BY t.LATENCY_MS) OVER (PARTITION BY a.APINAME) AS p99 FROM OTAEVisa_ESB_Logs.dbo.TRANSACTIONLOG t JOIN API a ON a.APIID=t.APIID WHERE t.CREATEDTIMESTAMP >= @from;

Top Error Codes TRANSACTIONLOG.ERRORCODE

SELECT TOP 10 ERRORCODE, COUNT(*) c FROM TRANSACTIONLOG WHERE HTTP_STATUS >= 400 AND CREATEDTIMESTAMP >= @from GROUP BY ERRORCODE ORDER BY c DESC;

Top DMCs by Package Volume PACKAGE_MASTER · DMC

SELECT TOP 10 d.DMC_NAME_EN, COUNT(p.PACKAGE_ID) packages, SUM(p.TOTAL_RECORDS) records FROM PACKAGE_MASTER p JOIN DMC d ON d.DMC_RECORDID = p.DMC_RECORDID WHERE p.CREATEDTIMESTAMP >= @from GROUP BY d.DMC_NAME_EN ORDER BY packages DESC;

Pass vs Fail Records PACKAGE_MASTER.TOTAL_PASSED_RECORDS / TOTAL_FAILED_RECORDS

SELECT CAST(CREATEDTIMESTAMP AS date) d, SUM(TOTAL_PASSED_RECORDS) passed, SUM(TOTAL_FAILED_RECORDS) failed FROM PACKAGE_MASTER WHERE CREATEDTIMESTAMP >= DATEADD(DAY,-13,GETDATE()) GROUP BY CAST(CREATEDTIMESTAMP AS date) ORDER BY d;

DMC Performance Leaderboard last 7d

DMCPackagesAppsPass %Avg LatencyStatus
SELECT d.DMC_NAME_EN, COUNT(p.PACKAGE_ID) pkgs, SUM(p.TOTAL_RECORDS) apps, 1.0*SUM(p.TOTAL_PASSED_RECORDS)/NULLIF(SUM(p.TOTAL_RECORDS),0) pass_rate, AVG(t.LATENCY_MS) avg_latency FROM PACKAGE_MASTER p JOIN DMC d ON d.DMC_RECORDID=p.DMC_RECORDID LEFT JOIN TRANSACTIONLOG t ON t.PACKAGE_ID=p.PACKAGE_ID WHERE p.CREATEDTIMESTAMP >= DATEADD(DAY,-7,GETDATE()) GROUP BY d.DMC_NAME_EN ORDER BY apps DESC;
